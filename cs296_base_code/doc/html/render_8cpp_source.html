162 void debug_draw_t::DrawString(int x, int y, const char *string, ...)
163 {
164  char buffer[128];

175  int h = glutGet(GLUT_WINDOW_HEIGHT);

178  glPushMatrix();
179  glLoadIdentity();
180 
181  glColor3f(0.9f, 0.6f, 0.6f);

183  int32 length = (int32)strlen(buffer);
184  for (int32 i = 0; i < length; ++i)
185  {
186  glutBitmapCharacter(GLUT_BITMAP_8_BY_13, buffer[i]);
187  }
188 
189  glPopMatrix();
190  glMatrixMode(GL_PROJECTION);
191  glPopMatrix();
192  glMatrixMode(GL_MODELVIEW);
193 }
