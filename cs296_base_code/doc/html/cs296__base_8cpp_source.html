55 void base_sim_t::pre_solve(b2Contact* contact, const b2Manifold* oldManifold)
56 {
57  const b2Manifold* manifold = contact->GetManifold();
58 
59  if (manifold->pointCount == 0)
60  {
61  return;
62  }
63 
64  b2Fixture* fixtureA = contact->GetFixtureA();
65  b2Fixture* fixtureB = contact->GetFixtureB();
66 
67  b2PointState state1[b2_maxManifoldPoints], state2[b2_maxManifoldPoints];
68  b2GetPointStates(state1, state2, oldManifold, manifold);
69 
70  b2WorldManifold world_manifold;
71  contact->GetWorldManifold(&world_manifold);
72 
73  for (int32 i = 0; i < manifold->pointCount && m_point_count < k_max_contact_points; ++i)
74  {
75  contact_point_t* cp = m_points + m_point_count;
76  cp->fixtureA = fixtureA;
77  cp->fixtureB = fixtureB;
78  cp->position = world_manifold.points[i];
79  cp->normal = world_manifold.normal;
80  cp->state = state2[i];
81  ++m_point_count;
82  }
83 }
