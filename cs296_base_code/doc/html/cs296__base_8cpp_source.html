00056 {
00057   const b2Manifold* manifold = contact->GetManifold();
00058   
00059   if (manifold->pointCount == 0)
00060     {
00061       return;

00064   b2Fixture* fixtureA = contact->GetFixtureA();

00066   
00067   b2PointState state1[b2_maxManifoldPoints], state2[b2_maxManifoldPoints];
00068   b2GetPointStates(state1, state2, oldManifold, manifold);
00069   
00070   b2WorldManifold world_manifold;
00071   contact->GetWorldManifold(&world_manifold);
00072   
00073   for (int32 i = 0; i < manifold->pointCount && m_point_count < k_max_contact_points; ++i)
00074     {
00075       contact_point_t* cp = m_points + m_point_count;
00076       cp->fixtureA = fixtureA;
00077       cp->fixtureB = fixtureB;
00078       cp->position = world_manifold.points[i];
00079       cp->normal = world_manifold.normal;
00080       cp->state = state2[i];
00081       ++m_point_count;
00082     }
00083 }
