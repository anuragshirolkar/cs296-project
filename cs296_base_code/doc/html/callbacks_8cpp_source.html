00043   b2Vec2 callbacks_t::convert_screen_to_world(int32 x, int32 y)
00044   {
00045     float32 u = x / static_cast<float32>(tw);
00046     float32 v = (th - y) / float32(th);
00047     
00048     float32 ratio = static_cast<float32>(tw) / static_cast<float32>(th);
00049     b2Vec2 extents(ratio * 25.0f, 25.0f);
00050     extents *= view_zoom;
00051     
00052     b2Vec2 lower = settings.view_center - extents;
00053     b2Vec2 upper = settings.view_center + extents;
00054   
00055     b2Vec2 p;
00056     p.x = (1.0f - u) * lower.x + u * upper.x;
00057     p.y = (1.0f - v) * lower.y + v * upper.y;
00058     return p;
