126  //! This initializes GLUT
127  glutInit(&argc, argv);
128  glutInitDisplayMode(GLUT_RGBA | GLUT_DOUBLE);
129  glutInitWindowSize(width, height);
130 
131  char title[50];
132  sprintf(title, "CS296 Base Code. Running on Box2D %d.%d.%d", b2_version.major, b2_version.minor, b2_version.revision);
133  main_window = glutCreateWindow(title);
134 
135  //! Here we setup all the callbacks we need

139  GLUI_Master.set_glutSpecialFunc(callbacks_t::keyboard_special_cb);
140  GLUI_Master.set_glutMouseFunc(callbacks_t::mouse_cb);
141  //! Others are set directly

144  glutKeyboardUpFunc(callbacks_t::keyboard_up_cb);
145  glutTimerFunc(frame_period, callbacks_t::timer_cb, 0);
146 
147  //! We create the GLUI user interface
